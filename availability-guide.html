function getAPIStatus(availability) { switch(availability) { case 'readily': return 'READY - Can use immediately'; case 'after-download': return 'READY - Will download model on first use'; case 'no': return 'UNAVAILABLE - API not supported'; default: return 'UNKNOWN - Unexpected response'; } }

Step 4: Complete Implementation

Here's the complete function used in the Chrome AI Shell:

async function checkAPIStatus() { const apis = ['LanguageModel', 'Translator', 'Summarizer', 'Writer', 'Rewriter', 'Proofreader']; for (const apiName of apis) { try { if (apiName in self) { let availability; if (apiName === 'Translator') { // Translator needs language parameters availability = await self[apiName].availability({ sourceLanguage: 'en', targetLanguage: 'es' }); } else { // Other APIs don't need parameters availability = await self[apiName].availability(); } console.log(`${apiName}: ${availability}`); } else { console.log(`${apiName}: Not found in browser`); } } catch (error) { console.log(`${apiName}: Error - ${error.message}`); } } }
